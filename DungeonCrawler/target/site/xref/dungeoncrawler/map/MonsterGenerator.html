7   import dungeoncrawler.creature.Creature;

14  public class MonsterGenerator {
15      
16      public List<Creature> getMonstersForLevel(LevelData levelData) {
17          List<Creature> monsters = new ArrayList<>();
18          int monsterAmount = levelData.getNumberOfMonstersOnLevel();
19          for (int i = 0; i < monsterAmount; i++) {
20              NPC newMonster = createRandomMonster(levelData);
21              monsters.add(newMonster);
22          }
23          return monsters;
24      }
25      
26      public NPC createRandomMonster(LevelData levelData) {
27          Random rng = new Random();
28          float difficultymultiplier = levelData.getMonsterDifficultyMultiplier();
29          boolean isHostile = true;
30          if (rng.nextDouble() < 0.33) {
31              isHostile = false;
32          }
33          String name = getNameForMonster();
34          int aggroRadius = rng.nextInt(4) + 1;
35          int monsterHealth = (int) ((rng.nextInt(51) + 50) * difficultymultiplier);
36          int monsterDamage = (int) ((rng.nextInt(5) + 6) * difficultymultiplier);
37          String symbol = "m";
38          if (!isHostile) {
39              symbol = "n";
40          }
41          NPC monster = new NPC(name, symbol);
42          monster.setHostileStatus(isHostile);
43          monster.setAggroRadius(aggroRadius);
44          monster.setMaxHealth(monsterHealth);
45          monster.setCurrentHealth(monsterHealth);
46          monster.setDamage(monsterDamage);
47          return monster;
48      }
49      
50      private String getNameForMonster() {
51          return "monster";
52      }
